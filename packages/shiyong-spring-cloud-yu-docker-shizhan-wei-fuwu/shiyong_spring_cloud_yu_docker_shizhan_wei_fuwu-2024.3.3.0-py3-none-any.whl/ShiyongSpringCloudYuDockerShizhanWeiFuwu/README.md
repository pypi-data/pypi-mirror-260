```
docker pull apachecn0/shiyong-spring-cloud-yu-docker-shizhan-wei-fuwu
docker run -tid -p <port>:80 apachecn0/shiyong-spring-cloud-yu-docker-shizhan-wei-fuwu
# 访问 http://localhost:{port} 查看文档
```
